139 AABCube OctreeNode<T>::getChildBounds(int index) const
140 {
141  assert(index >= 0 && index < 8);
142 
143  // If child exists, return its bounds.
144  if (m_children[index])
145  {
146  return m_children[index]->m_bounds;
147  }
148 
149  // Otherwise, compute them from the current parent node.
150  AABCube childBounds;
151  childBounds.halfWidth = m_bounds.halfWidth * 0.5f;
152  childBounds.center = m_bounds.center;
153  childBounds.center.x += ( (index & 1) ? 1.0f : -1.0f ) * childBounds.halfWidth;
154  childBounds.center.y += ( (index & 2) ? 1.0f : -1.0f ) * childBounds.halfWidth;
155  childBounds.center.z += ( (index & 4) ? 1.0f : -1.0f ) * childBounds.halfWidth;

161 void OctreeNode<T>::walkInsert(T &element,
162  Intersect intersect,
163  int depth,
164  int maxDepth,
165  float maxFill)
166 {
167  if (isLeaf())
168  {
169  // Should this leaf be subdivided?
170  const float depthFillRatio = m_elements.size() / (float) (1+depth);
171  const bool shouldSubdivide = depthFillRatio > maxFill &&
172  depth < maxDepth;
173  if (shouldSubdivide)
174  {
175  // Yes, subdivide this leaf..
176  m_isLeaf = false;
177  // .. and push elements to children.
178  while (!m_elements.empty())
179  {
180  walkInsert(m_elements.back(), intersect, depth, maxDepth, maxFill);
181  m_elements.pop_back();
182  }
183  walkInsert(element, intersect, depth, maxDepth, maxFill);
184  }
185  else
186  {
187  // No, populate this leaf and return.
188  m_elements.push_back(element);
189  }
190  return;
191  }
192 
193  // Not a leaf: locate children overlapping the element to insert.
194  int childIndex = 0;
195  for (auto &child: m_children)
196  {
197  const auto childBounds(getChildBounds(childIndex));
198  if (intersect(childBounds, element))
199  {
200  // Create child if needed.
201  if (!child)
202  {
203  child = new OctreeNode(childBounds);
204  assert(child);
205  }
206  // Walk down the tree under this child.
207  child->walkInsert(element, intersect, depth+1, maxDepth, maxFill);
208  }
209  ++childIndex;
210  }
211 }
