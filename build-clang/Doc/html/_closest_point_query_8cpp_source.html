65  const Point &vertex1,

67  const Point &fromPoint)
68 {
69  const Float3 edge0 = vertex1 - vertex0;
70  const Float3 edge1 = vertex2 - vertex0;
71  const Float3 v0 = vertex0 - fromPoint;
72 
73  const float a = edge0.dot(edge0);
74  const float b = edge0.dot(edge1);
75  const float c = edge1.dot(edge1);
76  const float d = edge0.dot(v0);
77  const float e = edge1.dot(v0);
78 
79  const float det = a*c - b*b;
80  const float s1 = b*e - c*d;
81  const float t1 = b*d - a*e;
82 
83  if (det == 0.0f)
84  {
85  throw std::invalid_argument("Collinear triangle vertices");
86  }
87 
88  float s2 = s1;
89  float t2 = t1;
90  if (s1 + t1 <= det)
91  {
92  if (s1 < 0.0f)
93  {
94  if (t1 < 0.0f)
95  {
96  // Region 4
97  if (d < 0.0f)
98  {
99  t2 = 0.0f;
100  if (-d >= a)
101  s2 = 1.0f;
102  else
103  s2 = -d/a;
104  }
105  else
106  {
107  s2 = 0.0f;
108  if (e >= 0.0f)
109  t2 = 0.0f;
110  else if (-e >= c)
111  t2 = 1.0f;
112  else
113  t2 = -e/c;
114  }
115  }
116  else
117  {
118  // Region 3
119  s2 = 0.0f;
120  if (e >= 0.0f)
121  t2 = 0.0f;
122  else if (-e >= c)
123  t2 = 1.0f;
124  else
125  t2 = -e/c;
126  }
127  }
128  else if (t1 < 0.0f)
129  {
130  // Region 5
131  t2 = 0.0f;
132  if (d >= 0.0f)
133  s2 = 0.0f;
134  else if (-d >= a)
135  s2 = 1.0f;
136  else
137  s2 = -d/a;
138  }
139  else
140  {
141  // Region 0
142  const float invDet = 1.0f / det;
143  s2 *= invDet;
144  t2 *= invDet;
145  }
146  }
147  else
148  {
149  if (s1 < 0.0f)
150  {
151  // Region 2
152  const float tmp0 = b+d;
153  const float tmp1 = c+e;
154  if (tmp1 > tmp0)
155  {
156  const float num = tmp1 - tmp0;
157  const float denom = a - 2.0f*b + c;
158  if (num >= denom)
159  s2 = 1.0f;

163  }
164  else

171  else

174  }
175  else if (t1 < 0.0f)
176  {
177  // Region 6
178  const float tmp0 = b + e;
179  const float tmp1 = a + d;
180  if (tmp1 > tmp0)
181  {
182  const float num = tmp1 - tmp0;
183  const float denom = a - 2.0f * b + c;
184  if (num >= denom)
185  t2 = 1.0f;
186  else
187  t2 = num/denom;
188  s2 = 1 - s2;
189  }
190  else
191  {
192  t2 = 0.0f;
193  if (tmp1 <= 0.0f)
194  s2 = 1.0f;
195  else if (d >= 0.0f)
196  s2 = 0.0f;
197  else
198  s2 = -d/a;
199  }
200  }
201  else
202  {
203  // Region 1
204  const float num = c + e - b - d;
205  if (num <= 0.0f)
206  s2 = 0.0f;
207  else
208  {
209  const float denom = a - 2.0f * b + c;
210  if (num >= denom)
211  s2 = 1.0;
212  else
213  s2 = num/denom;
214  }
215  t2 = 1.0f - s2;
216  }
217  }
218  const Point closestPoint = vertex0 + edge0*s2 + edge1*t2;
219  const float sqrDistance = (fromPoint - closestPoint).sqrLength();
220  return ClosestPointSpec(closestPoint, sqrDistance);

339 inline Point ClosestPointQuery::Impl::processMesh(const Point& queryPoint,
340  const float sqrMaxDist) const
341 {
342  const auto &vertices = m_vertices;
343  const auto reduceClosestPointOnFace = [&](const ClosestPointSpec &inputClosest,
344  const Face &face)
345  {
346  // Compute the closest point on this face and return the closest of this
347  // or the input, respecting sqrMaxDist.
348  const auto faceClosest = computeClosestPointOnFace(face, vertices, queryPoint);

359 void ClosestPointQuery::Impl::partitionSpace()
360 {
361  // Compute the extent of the space taken by all vertices.
362  Extent meshExtent = std::accumulate(m_vertices.begin(),
363  m_vertices.end(),
364  Extent(Point(infinity), Point(-infinity)),
365  growExtent);
366 
367  // Construct the root octree node bounding the mesh.
368  m_partitionedSpace = std::unique_ptr<Node>(new Node( computeCubicBounds(meshExtent) ));
369  auto& rootNode = *m_partitionedSpace;
370 
371  // Function that inserts a face into the octree.
372  const auto &vertices = m_vertices;
373  const auto insertFace = [&vertices, &rootNode](const Face &face)
374  {
375  const auto growFaceExtent = [&vertices](const Extent &extent,
376  const int vertexId)
377  {
378  return growExtent(extent, vertices[vertexId]);
379  };
380  // Compute extent of face vertices.
381  Extent faceExtent = std::accumulate(face.vertexIds.begin(),
382  face.vertexIds.end(),
383  Extent(Point(infinity), Point(-infinity)),
384  growFaceExtent);
385  // Insert this face into the octree.
386  rootNode.insert(OctreeElement(&face, computeBounds(faceExtent)),
387  intersect);
388  };
389  // Insert all faces into the octree.
390  std::for_each(m_faces.begin(), m_faces.end(), insertFace);
391 }

394 inline Point ClosestPointQuery::Impl::processPartitionedSpace(const Point& queryPoint,
395  const float sqrMaxDist) const
396 {
397  // Initialize the result.
398  auto result = ClosestPointSpec(Point(nan), infinity);
399 
400  // Initialize a heap whose top is the node closest to queryPoint.
401  using HeapEntry = std::pair<std::reference_wrapper<const Node>, float>;
402  const auto heapCompare = [&queryPoint](const HeapEntry &a, const HeapEntry &b)
403  {
404  return (a.second > b.second);
405  };
406  using HeapContainer = std::vector<HeapEntry>;
407  using HeapCompareType = decltype(heapCompare);
408  using Heap = std::priority_queue< HeapEntry,
409  HeapContainer,
410  HeapCompareType >;
411  Heap heap(heapCompare);
412 
413  // Prepare octree visitor functions.
414  const auto &vertices = m_vertices;
415  // When visiting an element (face)..
416  const auto visitElement = [&](const OctreeElement &element)
417  {
418  // .. compute the closest point to it and update the global result,
419  // respecting sqrMaxDist.
420  assert(element.first);
421  const auto &face = *(element.first);
422  const auto faceClosest = computeClosestPointOnFace(face, vertices, queryPoint);
423  if (faceClosest.second < sqrMaxDist && faceClosest.second < result.second)
424  result = faceClosest;
425  };
426 
427  // When visiting an octree child..
428  const auto visitChild = [&queryPoint, &heap, &result](Node const &child)
429  {
430  // ..if the child is closer than the closest point currently found..
431  const float nodeSqrDist = computeSqrDistanceToBounds( queryPoint,
432  child.getBounds() );
433  if (nodeSqrDist < result.second)
434  {
435  //.. then add it to the heap.
436  heap.push( HeapEntry(child, nodeSqrDist) );
437  }
438  };
439 
440  // Initialize the heap with the octree root.
441  const auto &rootNode = *m_partitionedSpace;
442  const float rootSqrDist = computeSqrDistanceToBounds( queryPoint,
443  rootNode.getBounds() );
444  heap.push( HeapEntry(std::cref(rootNode), rootSqrDist) );
445 
446  // While the heap has nodes and the top one is closer than the current result,
447  while (!heap.empty() && heap.top().second < result.second)
448  {
449  // Eat the top of the heap.
450  const Node &node = heap.top().first.get();
451  heap.pop();
452 
453  if (node.isLeaf())
454  {
455  // If it's a leaf, visit the elements (faces).
456  node.accept(visitElement);
457  }
458  else
459  {
460  // Otherwise, visit the children nodes.
461  node.accept(visitChild);
462  }
463  }
464 
465  return result.first;
466 }
